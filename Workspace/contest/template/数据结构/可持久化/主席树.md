struct HJTTree {
    struct Node {
        int l, r;
        int sum;
    };

    vector<Node> tr;
    vector<int> root;   // 每个版本的根
    int n;

    HJTTree(int n) : n(n) {
        tr.push_back({0,0,0}); // 0号空节点
        root.push_back(build(1, n));
    }

    // ===== 新建节点（复制）=====
    int clone(int node) {
        tr.push_back(tr[node]);
        return (int)tr.size() - 1;
    }

    // ===== 建树 =====
    int build(int l, int r) {
        int node = tr.size();
        tr.push_back({0,0,0});

        if (l != r) {
            int mid = (l + r) >> 1;
            tr[node].l = build(l, mid);
            tr[node].r = build(mid+1, r);
        }
        return node;
    }

    // ===== 单点修改（产生新版本）=====
    int update(int pre, int l, int r, int pos, int val) {
        int node = clone(pre);

        if (l == r) {
            tr[node].sum += val;
            return node;
        }

        int mid = (l + r) >> 1;
        if (pos <= mid)
            tr[node].l = update(tr[pre].l, l, mid, pos, val);
        else
            tr[node].r = update(tr[pre].r, mid+1, r, pos, val);

        tr[node].sum =
            tr[tr[node].l].sum +
            tr[tr[node].r].sum;

        return node;
    }

    // 外部调用：生成新版本
    void modify(int version, int pos, int val) {
        root.push_back(
            update(root[version], 1, n, pos, val)
        );
    }

    // ===== 区间查询 =====
    int query(int node, int l, int r, int ql, int qr) {
        if (ql <= l && r <= qr)
            return tr[node].sum;

        int mid = (l + r) >> 1;
        int res = 0;

        if (ql <= mid)
            res += query(tr[node].l, l, mid, ql, qr);
        if (qr > mid)
            res += query(tr[node].r, mid+1, r, ql, qr);

        return res;
    }

    int range_query(int version, int l, int r) {
        return query(root[version], 1, n, l, r);
    }

    // ===== 区间第k小 =====
    // u = 左版本 root
    // v = 右版本 root
    int kth(int u, int v, int l, int r, int k) {
        if (l == r) return l;

        int cnt =
            tr[tr[v].l].sum -
            tr[tr[u].l].sum;

        int mid = (l + r) >> 1;

        if (k <= cnt)
            return kth(tr[u].l, tr[v].l, l, mid, k);
        else
            return kth(tr[u].r, tr[v].r, mid+1, r, k-cnt);
    }
};

void solve() {
    int n; cin >> n;
    vector<int> a(n);

    //离散化
    vector<int> b = a;
    sort(b.begin(), b.end());
    b.erase(unique(b.begin(), b.end()), b.end());

    auto get = [&](int x) -> int {
        return lower_bound(b.begin(), b.end(), x) - b.begin() + 1;
    };

    //建树
    HJTTree pst(b.size());

    for(int i = 1; i <= n; i++) {
        pst.modify(i-1, get(a[i]), 1);
    }

    //查询 [l, r] 内第k小
    int l, r, k; cin >> l >> r >> k;

    int id = pst.kth(pst.root[l-1], pst.root[r], 1, b.size(), k);

    cout << b[id-1];

}
