struct BIT {
    #define lowbit(x) (x & (-x))

    int n;
    vector<int> tree;

    BIT (int n) {
        this->n = n;
        tree.resize(n + 1);
    }

    void add(int x, int a) {
        for (int i = x; i <= n; i += lowbit(i)) {
            tree[i] += a;
        }
    }

    int sum(int x) {
        int res = 0;
        for (int i = x; i; i -= lowbit(i)) {
            res += tree[i];
        }
        return res;
    }

    int rangeSum(int l, int r) {
        return sum(r) - sum(l - 1);
    }
};
