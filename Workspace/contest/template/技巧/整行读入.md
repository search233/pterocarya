vector<ll> read() {
    string s;
    getline(cin >> ws, s);

    vector<ll> res;

    ll num = 0;
    int sign = 1;
    bool in = false;   // 当前是否正在读数字

    for (char c : s) {
        if (c == '-') {
            sign = -1;
            in = true;
            num = 0;
        }
        else if (isdigit(c)) {
            num = num * 10 + (c - '0');
            in = true;
        }
        else {
            if (in) {
                res.push_back(sign * num);
                num = 0;
                sign = 1;
                in = false;
            }
        }
    }

    // 行末数字
    if (in)
        res.push_back(sign * num);

    return res;
}
