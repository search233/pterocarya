struct Disc { //Discretization
    vector<int> v;

    Disc(const vector<int>& vec) {
        for (auto& num : vec) {
            v.push_back(num);
        }

        ranges::sort(v);
        v.erase(unique(v.begin(), v.end()), v.end());
    }

    vector<int> func() {
        return v;
    }

    int find(int num) {
        return lower_bound(v.begin(), v.end(), num) - v.begin() + 1;
    }
};
