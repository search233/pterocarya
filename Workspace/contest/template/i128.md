using i128 = __int128_t;

// 向上取整的除法：ceil(a / b)，b > 0
i128 div_ceil(i128 a, i128 b) {
    if (b < 0) a = -a, b = -b;
    if (a >= 0) return (a + b - 1) / b;
    else return a / b; // 负数时 C++ 向 0 截断，本身就是 ceil
}

// 向下取整的除法：floor(a / b)，b > 0
i128 div_floor(i128 a, i128 b) {
    if (b < 0) a = -a, b = -b;
    if (a >= 0) return a / b;
    else return (a - (b - 1)) / b;
}

// 输出重载
ostream& operator<<(ostream& os, i128 x) {
    if (x == 0) {
        os << '0';
        return os;
    }
    if (x < 0) {
        os << '-';
        x = -x;
    }
    string s;
    while (x > 0) {
        int digit = (int)(x % 10);
        s.push_back('0' + digit);
        x /= 10;
    }
    reverse(s.begin(), s.end());
    os << s;
    return os;
}

// 输入重载
istream& operator>>(istream& is, i128& x) {
    string s;
    is >> s;
    x = 0;
    bool neg = false;
    int i = 0;
    if (s[0] == '-') {
        neg = true;
        i = 1;
    }
    for (; i < (int)s.size(); ++i) {
        x = x * 10 + (s[i] - '0');
    }
    if (neg) x = -x;
    return is;
}
